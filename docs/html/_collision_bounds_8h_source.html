42  template <typename _Points> static golem::BoundingBox::Desc getBoundingBox(_Points points) {
43  golem::BoundingBox::Desc boundingBoxDesc;
44  boundingBoxDesc.pose.R = orientation(covariance(points));
45  golem::Vec3 v[3];
46  for (size_t j = 0; j < 3; ++j) {
47  v[j].setZero();
48  v[j][j] = golem::REAL_ONE;
49  v[j] = boundingBoxDesc.pose.R * v[j];
50  }
51  golem::Vec3 point, min(golem::REAL_MAX), max(-golem::REAL_MAX);
52  for (size_t i = 0; points(i, point); ++i) {
53  for (size_t j = 0; j < 3; ++j) {
54  const golem::Real proj = point.dot(v[j]);
55  min[j] = std::min(min[j], proj);
56  max[j] = std::max(max[j], proj);
57  }
58  }
59  for (size_t j = 0; j < 3; ++j) {
60  boundingBoxDesc.dimensions[j] = golem::REAL_HALF*(max[j] - min[j]);
61  boundingBoxDesc.pose.p[j] = min[j] + boundingBoxDesc.dimensions[j];
62  }
63  boundingBoxDesc.pose.p = boundingBoxDesc.pose.R * boundingBoxDesc.pose.p;
64  return boundingBoxDesc;
65  }
