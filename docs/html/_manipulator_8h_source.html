276  template <typename _Iter> static void bounds(_Iter begin, _Iter end, golem::Real distance, _Iter& lo, _Iter& hi) {
277  if (begin + 2 > end)
278  throw golem::Message(golem::Message::LEVEL_ERROR, "Manipulator::Waypoint::bounds(): At least two waypoints required");
279 
280  // binary search
281  hi = std::lower_bound(begin, end, distance, [] (const golem::Sample<golem::Real>& l, const golem::Real& r) {return l.weight < r; });
282  if (hi == end)
283  --hi;

285  ++hi;
286  lo = hi - 1;
287  }

291  if (begin + 1 > end)

459  template <typename _Iter, typename _Find> std::pair<_Iter, RBDistEx> find(_Iter begin, _Iter end, _Find find) const {
460  if (begin == end)
461  throw golem::Message(golem::Message::LEVEL_ERROR, "Manipulator::find(): At least one waypoint required");
462 
463  golem::Real err = golem::REAL_MAX;
464  RBDistEx dist;
465  _Iter j = begin;
466  for (_Iter i = begin; i != end; ++i) {
467  const RBDistEx newDist = find(i);
468  const golem::Real newErr = desc.trajectoryErr.dot(newDist);
469  if (err >= newErr && !newDist.collision || i == begin) {
470  err = newErr;
471  dist = newDist;
472  j = i;
473  }
474  if (newErr < golem::REAL_ONE && !newDist.collision)
475  return std::make_pair(i, newDist);
476  }
477 
478  if (desc.trajectoryThrow)
479  throw golem::Message(golem::Message::LEVEL_ERROR, "Manipulator::find(): No trajectory found");
480 
481  return std::make_pair(j, dist);
