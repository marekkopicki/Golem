381  //const Real b = Real(0.66666666666666666667)*(Math::sqrt(Math::sqr(k) + Real(2.25)) - k); // DIM=4, 2/3*(sqrt(k^2 + 9/4) - k)
382  const Real b = Math::sqrt(Math::sqr(k) + numeric_const<Real>::ONE) - k; // DIM=3, sqrt(k^2 + 1) - k
383  const Real x = (numeric_const<Real>::ONE - b) / (numeric_const<Real>::ONE + b);
384  //const Real c = k*x + Real(3.0)*Math::ln(numeric_const<Real>::ONE - Math::sqr(x)); // DIM=4

386 
387  for (;;) {
388  // beta distribution with a = b = (DIM - 1)/2 = 1, where DIM=3
389  const Real o = numeric_const<Real>::TWO*rand.template nextUniform<Real>() - numeric_const<Real>::ONE;
390  const Real p = rand.template nextUniform<Real>();
391  const Real s = Math::sqr(o) + Math::sqr(p);
392  if (s > numeric_const<Real>::ONE)
393  continue;
394  const Real z = numeric_const<Real>::HALF + o*p*Math::sqrt(numeric_const<Real>::ONE - s) / s;
395

398  const Real w = (numeric_const<Real>::ONE - z*(numeric_const<Real>::ONE + b)) / v;
399  //const Real t = k*w + Real(3.0)*Math::ln(numeric_const<Real>::ONE - x*w) - c; // DIM=4

401  if (t < Math::ln(u))
402  continue;
403

405  this->z = w;
