231  template <typename _Real> inline _Real N(const _Real x, const _Real stdev) {
232  const _Real norm = golem::numeric_const<_Real>::ONE / (stdev*golem::Math::sqrt(2 * golem::numeric_const<_Real>::PI));
233  return norm*golem::Math::exp(-.5*golem::Math::sqr(_Real(x) / _Real(stdev))); // gaussian
234  };
235  // computes guassian on a vector
236  template <typename _Ptr, typename _Real> inline std::vector<_Real> N(_Ptr begin, _Ptr end, const size_t dim, const _Real stdev) {
237  std::vector<_Real> output;
238  output.assign(dim, golem::numeric_const<_Real>::ZERO);
239  size_t idx = 0;
240  for (_Ptr i = begin; i != end; ++i) {
241  output[idx++] = N(*i, stdev);
242  }
243  return output;
244  };
