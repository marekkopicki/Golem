260  inline void makeLogNormProduct(size_t activeContacts, golem::Real penaltyExp) {
261 
262  valueActive = golem::REAL_ZERO;
263  valueInactive = golem::REAL_ZERO;
264  size_t n = 0;
265 
266  for (size_t i = 0; i < Manipulator::Link::RealCoord::SIZE; ++i) {
267  const golem::Real contact = contacts[i];
268  valueActive += getLogValue(contact);
269  if (contact > golem::REAL_ZERO) {
270  valueInactive += contact;
271  ++n;
272  }
273  }
274 
275  valueInactive = n > 0 && activeContacts > n ? getLogValue(valueInactive/n)*(activeContacts - n)*penaltyExp : golem::REAL_ZERO;
276 
277  value = golem::REAL_ZERO;
278  value += valueActive;
279  value += valueInactive;
