344  inline void generate(Rand& rand, const VecSeq& vectorSeq, Vec& nextVector, size_t& nextIndex) const {
345  // choose random set of vectors (use random permutation for small population size)
346  size_t index[4];
347  for (size_t i = 0; i < 4;) {
348  REPEAT:
349  index[i] = rand.next()%vectorSeq.size();
350  for (size_t j = 0; j < i; ++j)
351  if (index[j] == index[i])
352  goto REPEAT;
353  ++i;
354  }
355  nextIndex = index[0];
356 
357  // generate a candidate solution
358  const size_t crossPoint = rand.next()%heuristic.size();
359  bool bCross = true;
360  for (size_t i = 0; i < heuristic.size(); ++i) {
361  const size_t j = (crossPoint + i)%heuristic.size();
362  if (bCross) {
363  // apply crossover accessor
364  nextVector[j] = vectorSeq[index[1]][j] + desc.diffFac*(vectorSeq[index[2]][j] - vectorSeq[index[3]][j]);
365  // decide when to stop crossover
366  bCross = desc.crossProb < rand.nextUniform<Type>();
367  }
368  else {
369  // just copy if crossover or coordinate is disabled
370  nextVector[j] = vectorSeq[index[0]][j];
371  }
372  }
373  }
375  inline bool accept(Type prevValue, Type nextValue) const {
376  return desc.minimum ? prevValue > nextValue : prevValue < nextValue;
377  }
379  inline Type variance(const ValSeq& valueSeq) const {

385  inline Type variance(Rand& rand, const VecSeq& vectorSeq) const {
386  // population size
387  const size_t size = std::min(vectorSeq.size(), generation);
388  // population variance
389  Variable<Type> variable;
390  for (size_t i = 0; i < desc.testPairs; ++i) {
391  const size_t j = rand.next()%size;
392  const size_t k = (j + 1 + rand.next()%(size - 1))%size;
393  variable.update(heuristic.distance(vectorSeq[j], vectorSeq[k]));
394  }
395  return variable.getVarianceN();
396  }
398  inline bool stop(Rand& rand, const ValSeq& valueSeq, const VecSeq& vectorSeq, Type& vectorVar) const {
399  if (generation >= desc.generationsNum || generation%desc.testGenerations == 0)
400  return (vectorVar = desc.testValue ? variance(valueSeq) : variance(rand, vectorSeq)) < desc.testVariance || generation >= desc.generationsNum;
401  return heuristic.getContext().getTimer().elapsed() > runStop;
402  }

405  void run() {
406  Vec vector; // no initialisation - make sure that both sample() and generate() initialise *ALL* coordinates (including disabled ones!)
407  Type value;
408  size_t index;
409  ThreadData threadData(heuristic.threadData());
410  Rand rand(RandSeed(context.getRandSeed()._U32[0] + (desc.numOfThreads > 0 ? pParallels->getCurrentJob()->getJobId() : (U32)0), (U32)0));
411 
412  // thread initialisation
413  heuristic.init(threadData, rand);
414 
415  for (bool bAccept = false;;) {
416  {
417  CriticalSectionWrapper csw(cs, desc.numOfThreads > 0);
418 
419  // increase generation index
420  if (generation >= vectorSeq.size() || bAccept)
421  generation += 1;
422 
423  // check if the next solution is better than the previous one
424  if (bAccept) {
425  vectorSeq[index] = vector;
426  valueSeq[index] = value;
427  bAccept = false;
428  }
429 
430  // stop condition - the maximum number of generations or passed test
431  if (bStop || generation > 0 && stop(rand, valueSeq, vectorSeq, vectorVar)) {
432  bStop = true; // stop other threads
433  break;
434  }
435 
436  if (generation < vectorSeq.size()) {
437  // sample test vector
438  bAccept = heuristic.sample(rand, vector, value);
439  index = generation;
440  }
441  else {
442  // generate test vector
443  generate(rand, vectorSeq, vector, index);
444  value = valueSeq[index]; // get old value corresponding to index
445  }
446  }
447 
448  // compute value of test vector
449  heuristic.process(vector, threadData);
450  const Type nextValue = heuristic.value(vector, threadData);
451  // accept if there is no collision
452  if (!bAccept && accept(value, nextValue) && !heuristic.collides(vector, threadData))
453  bAccept = true;
454  // always update here!
455  value = nextValue;
456  }
457  }
