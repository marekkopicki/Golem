542  /* Returns dest pointers to the range of items begin, ..., end such that:
543  * *destBegin <= *srcBegin, *(destBegin + 1) > *srcBegin, ..., *(destEnd - 1) <= *srcEnd, *destEnd > *srcEnd
544  */
545  template <typename _Ptr, typename _Cmp> static void lookup(_Ptr begin, _Ptr end, _Cmp srcBegin, _Cmp srcEnd, _Ptr& destBegin, _Ptr& destEnd) {
546  if (begin == end) {
547  destBegin = destEnd = end;
548  return;
549  }
550 
551  destBegin = std::lower_bound(begin, end, srcBegin);
552  if (destBegin != begin)
553  --destBegin;
554  destEnd = std::upper_bound(begin, end, srcEnd);
555  if (destEnd != end)
556  ++destEnd;
557  }
558 
559  /* Returns dest pointers to the range of items containing src
560  */
561  template <typename _Ptr, typename _Cmp> static bool lookup(_Ptr begin, _Ptr end, _Cmp src, _Ptr& destBegin, _Ptr& destEnd) {
562  lookup(begin, end, src, src, destBegin, destEnd);
563 
564  if (destBegin == destEnd)
565  return false;
566 
567  --destEnd;
568
