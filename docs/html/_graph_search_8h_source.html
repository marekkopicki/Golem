112  template <class _Heuristic> U32 find(_Heuristic& heuristic) {
113  // the current node.
114  U32 nCurrent = IDX_ROOT;
115  // the node through which passes the best path to nCurrent.
116  U32 rCurrent = IDX_ROOT;
117  // a cost of reaching nCurrent.
118  Real cCurrent = Node::COST_ZERO;
119  // clear node cost rank
120  nodeRank.clear();
121 
122  for (;;) {
123  // The best path to nCurrent goes through rCurrent.
124  closedVec[nCurrent] = rCurrent;
125 
126  // Break the loop if nCurrent is the goal.
127  if (heuristic.goal(nCurrent))
128  break;
129 
130  // then remove nCurrent from open.
131  openVec[nCurrent] = Node::IDX_UINI;
132 
133  for (;;) {
134  // Expand the current node nCurrent,
135  // find for all possible destination nodes j
136  const U32 j = heuristic.expand(nCurrent);
137 
138  // Break if there are no more destination nodes
139  if (j == Node::IDX_UINI)
140  break;
141  // Do not expand if the destination node j is:
142  // equal the expanded node nCurrent, or is already on close.
143  if (j == nCurrent || closedVec[j] != Node::IDX_UINI)
144  continue;
145 
146  // Calculate the cost of reaching j through nCurrent.
147  Real cNew = heuristic.cost(nCurrent, j);
148 
149  // if j is reachable
150  if (cNew < Node::COST_INF) {
151  // Calculate the total cost
152  cNew += cCurrent;
153 
154  // Extract a node, through which passes the best path to j.
155  const U32 k = openVec[j];
156  // Redirect the best path to j if there was no
157  // path to j before, or the new path is less costly.
158  if (k == Node::IDX_UINI) {
159  if (heuristic.collision(nCurrent, j))
160  continue;
161  }
162  else {
163  const Real cOld = costMat[j];
164  if (cOld <= cNew || heuristic.collision(nCurrent, j))
165  continue;
166  nodeRank.erase(Node(j, cOld));
167  }
168 
169  // the best path to j passes nCurrent
170  openVec[j] = nCurrent;
171  // Update cost map and rank
172  costMat[j] = cNew;
173  nodeRank.insert(Node(j, cNew));
174  }
175  }
176 
177  // Get a node nCurrent with the lowest cost value.
178  NodeRank::iterator node = nodeRank.begin();
179  // Break the loop if nodeRank/open is empty - goal node is unreacheable.
180  if (node == nodeRank.end())
181  return Node::IDX_UINI;
182 
183  nCurrent = node->index;
184  rCurrent = openVec[nCurrent];
185  cCurrent = costMat[nCurrent];
186 
187  // remove nCurrent from nodeRank
188  nodeRank.erase(node);
189  }
190 
191  return nCurrent;
192  }
