78  const PlannerInfo& plannerInfo;
79  const golem::ConfigspaceCoord distance;
80  virtual golem::Real distConfigspaceCoord(const golem::ConfigspaceCoord& prev, const golem::ConfigspaceCoord& next) const {
81  golem::Real dist = golem::REAL_ZERO;
82  for (golem::Configspace::Index i = plannerInfo.armInfo.getJoints().begin(); i < plannerInfo.armInfo.getJoints().end(); ++i)
83  dist += distance[i] * golem::Math::sqr(prev[i] - next[i]);
84  for (golem::Configspace::Index i = plannerInfo.handInfo.getJoints().begin(); i < plannerInfo.handInfo.getJoints().end(); ++i)
85  dist += distance[i] * golem::Math::sqr(prev[i] - next[i]);
86  return golem::Math::sqrt(dist);
87  }
88  virtual golem::Real distCoord(golem::Real prev, golem::Real next) const { return golem::Math::abs(prev - next); }
89  virtual bool distCoordPlanning(const golem::Configspace::Index& index) const { return plannerInfo.armInfo.getJoints().contains(index) || plannerInfo.handInfo.getJoints().contains(index); }
90  virtual bool distCoordInterpolation(const golem::Configspace::Index& index) const { return plannerInfo.handInfo.getJoints().contains(index); }
91  public:
92  Profile(const PlannerInfo& plannerInfo, const golem::Profile::Desc& desc, const golem::ConfigspaceCoord& distance) : golem::Profile(plannerInfo.planner->getController()), plannerInfo(plannerInfo), distance(distance) {
