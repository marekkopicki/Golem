308  ++node.first;
309 
310  if (i < graphPartition) {
311  if (node.first >= graphSize)
312  node.first = Node::IDX_UINI;
313  }

330  if (i >= graphPartition && j >= graphPartition) {
331 #ifdef _PATHFINDER_PERFMON

333 #endif
334  return node.second;
335  }
336  else {
337 #ifdef _PATHFINDER_PERFMON
338  ++costOnline;
339 #endif
340  return heuristic.cost(graph[i], graph[j]);
341  }
342  }

402  if (!bidirectional)

407  forward = true;
408

418  }
419  else
420  run();
421 
422  // extract path
423  if (nodeIndex != Node::IDX_UINI) {
424  graphHeuristicForward.graphSearch.extract(nodeIndex, nodePath, nodePath.end());
425  Node::Seq tmp;
426  graphHeuristicBackward.graphSearch.extract(nodeIndex, tmp, tmp.end());
427  if (!tmp.empty()) {
428  tmp.pop_back(); // do not duplicate nodeIndex
429  nodePath.insert(nodePath.end(), tmp.rbegin(), tmp.rend()); // reverse order
430  }
431  }
432  }

434  void run() {
435  // choose current search direction
436  bool forward;
437  {
438  CriticalSectionWrapper csw(cs);
439  forward = this->forward;
440  this->forward = !this->forward;
441  }
442
