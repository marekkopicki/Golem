235  template <typename _Ptr> static size_t getSize(_Ptr begin, _Ptr end) {
236  size_t size = 0;
237  for (_Ptr i = begin; i != end; ++i)
238  size += i->size*i->count;

242  template <typename _Ptr> static void getOffset(Controller& controller, _Ptr begin, _Ptr end, size_t reservedBegin, size_t chainBegin, size_t jointBegin, Seq& seq) {
243  seq.clear();
244  idxdiff_t offset = 0;
245  for (_Ptr i = begin; i != end; ++i) {
246  const idxdiff_t size = i->size*i->count;
247  if (size > 0) {
248  idxdiff_t shift = reservedBegin;
249 
250  switch (i->type) {
251  case TYPE_CHAINSPACE:
252  shift -= chainBegin*i->size;
253  break;
254  case TYPE_CONFIGSPACE:
255  shift -= jointBegin*i->size;
256  break;
257  }
258 
259  controller.getContext().verbose("%s: reserved area {id=%s, type=%s, size=%d, rel_offset=%d, abs_offset=%d}, {reserved=%d, chain=%d, joint=%d}\n", controller.getID().c_str(), i->name.c_str(), TypeName[i->type].c_str(), size, offset, offset + shift, reservedBegin, chainBegin, jointBegin);
260 
261  seq.push_back(offset + shift);
262  offset += size;
263  }
264  else {
265  controller.getContext().verbose("%s: reserved area (not available) {id=%s, type=%s}\n", controller.getID().c_str(), i->name.c_str(), TypeName[i->type].c_str());
266 
267  seq.push_back(UNAVAILABLE);
268  }
269  }
