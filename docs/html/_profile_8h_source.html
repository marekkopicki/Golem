230  template <typename _Ptr> void create(const _Ptr pbegin, const _Ptr pend, const Controller::State& first, const Controller::State& last, Controller::State::Seq &seq, Controller::State::Seq::iterator& begin, Controller::State::Seq::iterator& end) const {
231  const size_t size = size_t(pend - pbegin);
232  if (size < 2)
233  throw MsgProfileSize(Message::LEVEL_ERROR, "Profile::create(): Insuffucient number of waypoints %u", size);
234 
235  for (_Ptr ptr = pbegin; ptr < pend; ++ptr) {
236  // first by default
237  Controller::State state = first;
238 
239  // interpolation
240  const Real s = Real(ptr - pbegin) / (size - 1), p[2] = { REAL_ONE - s, s };
241 
242  // time stamp
243  state.t = p[0] * first.t + p[1] * last.t;
244 
245  // coordinates
246  for (Configspace::Index n = controller.getStateInfo().getJoints().begin(); n < controller.getStateInfo().getJoints().end(); ++n)
247  if (pCallbackDist->distCoordPlanning(n)) {
248  state.cpos[n] = ptr->cpos[n];
249  state.cvel[n] = state.cacc[n] = REAL_ZERO;
250  }
251  else if (pCallbackDist->distCoordInterpolation(n)) {
252  state.cpos[n] = p[0] * first.cpos[n] + p[1] * last.cpos[n];
253  state.cvel[n] = state.cacc[n] = REAL_ZERO;
254  }
255 
256  begin = ++seq.insert(begin, state);
257  }
258 
259  end = begin;
260  begin = end - size;
261  }
