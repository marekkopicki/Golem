285  for (size_t i = 0; i < bounds.size(); ++i) {
286  const BoundingConvexMesh* mesh = dynamic_cast<const BoundingConvexMesh*>(bounds[i].get());
287  if (mesh != nullptr) {
288  surfaces.resize(surfaces.size() + 1);
289  triangles.resize(triangles.size() + 1);
290  surfaces.back().resize(mesh->getTriangles().size());
291  triangles.back().resize(mesh->getTriangles().size());
292  for (size_t j = 0; j < mesh->getTriangles().size(); ++j) {
293  triangles.back()[j].normal = surfaces.back()[j].normal = Vec3(mesh->getNormals()[j]);
294  surfaces.back()[j].t1 = Vec3(mesh->getVertices()[mesh->getTriangles()[j].t1]); // e.g. first triangle

328  static inline Real getDepth(const typename Triangle::Seq& triangles, const Vec3& point) {
329  Real depth = numeric_const<Real>::MAX;
330  for (typename Triangle::Seq::const_iterator i = triangles.begin(); i != triangles.end(); ++i) {
331  const Real d = i->distance - i->normal.dot(point);
332  if (d < numeric_const<Real>::ZERO) // no HBCollision
333  return d;
334  if (depth > d) // search for minimum
335  depth = d;
336  }
337  return depth;
338  }
340  inline Real getDepth(const Vec3& point, const bool distance = false) const {
341  //static size_t jj = 0;
342  // if distance true, and there are not HBCollisions, then returns the closest point outside the bounds (negative values)
343  Real depth = distance ? -numeric_const<Real>::MAX : numeric_const<Real>::ZERO; // if no bounds or HBCollisions, no effect
344  for (typename Triangle::SeqSeq::const_iterator i = triangles.begin(); i != triangles.end(); ++i) {
345  const Real d = getDepth(*i, point);
346  if (depth < d) // search for maximum depth (can collide only with one mesh within bounds)
347  depth = d;
348  }
349  //if (jj % 100 == 0) printf("getdepth=%f\n", depth);
350  return depth;
351  }

358  static inline Real getSurfaceDistance(const typename Triangle::Seq& triangles, const Vec3& point, const Face face) {
359  Real distance = numeric_const<Real>::MAX;
360  Real penetration = numeric_const<Real>::ONE;
361  for (typename Triangle::Seq::const_iterator i = triangles.begin(); i != triangles.end(); ++i) {
362  if (face != Face::UNKNOWN && i->face != face)
363  continue;
364  const Real d2 = i->point.distance(point);
365  if (distance > d2) // search for minimum distance
366  distance = d2;
367  // check for penetration
368  if (penetration > numeric_const<Real>::ZERO) {
369  const Real d = i->distance - i->normal.dot(point);
370  if (d > numeric_const<Real>::ZERO) // no HBCollision
371  penetration = -numeric_const<Real>::ONE;
372  }
373  }
374  return penetration*distance;
375  }

378  // penetration value is positive if there is a penetration
379 // const Real penetration = getDepth(point) > REAL_ZERO ? REAL_ONE : -REAL_ONE;

385  distance = d;
386  }

388  return /*penetration**/distance;

402  template <typename _Ptr> inline _RealEval distance(_Ptr begin, _Ptr end, Vec3& frame, U32& HBCollisions, const Face face = Face::UNKNOWN) const {
403  Vec3 inFrame, outFrame; inFrame.setZero(); outFrame.setZero();
404  _RealEval eval = numeric_const<_RealEval>::ZERO, c = numeric_const<_RealEval>::ZERO, minDist = -numeric_const<Real>::MAX;
405  for (_Ptr i = begin; i < end; ++i) {
406  const Real distance = getSurfaceDistance(*i, face);
407  if (distance > numeric_const<Real>::ZERO) {
408  kahanSum(eval, c, distance);
409  inFrame += (*i).getPoint()*frameWeight(distance);
410  ++HBCollisions;
411  }
412  else if (minDist < distance) {
413  minDist = distance;
414  outFrame = (*i).getPoint();
415  }
416  }
417  frame = HBCollisions > 0 ? inFrame : outFrame;
418  return HBCollisions > 0 ? eval/HBCollisions : minDist;
419  }

422  template <typename _Ptr> inline _RealEval evaluate(_Ptr begin, _Ptr end, _RealEval depthStdDev, size_t& HBCollisions) const {
423  _RealEval eval = numeric_const<_RealEval>::ZERO, c = numeric_const<_RealEval>::ZERO;
424  for (_Ptr i = begin; i < end; ++i) {
425  const Real depth = getDepth(*i);
426  if (depth > numeric_const<Real>::ZERO) {
427  const _RealEval pointEval = Math::exp(-depthStdDev*Real(depth));
428  kahanSum(eval, c, pointEval);
429  ++HBCollisions;
430  }
431  }
432  return eval;
433  }

436  template <typename _Ptr> inline _RealEval evaluate(_Ptr begin, _Ptr end, _RealEval depthStdDev, _RealEval distanceStdDev, bool observation, const Mat34 &pose, const Real &force, size_t& HBCollisions) const {
437  _RealEval eval = numeric_const<_RealEval>::EPS, c = numeric_const<_RealEval>::ZERO;
438  Real norm = numeric_const<Real>::ONE / (depthStdDev*Math::sqrt(2 * numeric_const<Real>::PI));
439  for (_Ptr i = begin; i < end; ++i) {
440  const Real depth = getDepth(*i, true); // penetration depth (positive values) or distance from the closest surfaces (negative values)
441  const bool direction = match(pose, *i, force);
442  // this hypothesis generate a contact that does not match the observation
443  if (!observation && depth > numeric_const<Real>::ZERO) {
444 // printf("wrong direction\n");
445  return numeric_const<_RealEval>::ZERO;
446  }
447  // compute likelihood only for bounds were a contact occurred
448  if (observation) {
449  const _RealEval pointEval = norm*Math::exp(-.5*Math::sqr(Real(depth)/Real(depthStdDev))); // gaussian
450  kahanSum(eval, c, pointEval);
451  // penalise concats not on the bound's surface
452  if (depth >= 0.0008 || !direction) {
453  const _RealEval penalise = -.5*pointEval; //-Math::exp(-Real(depth)); //-.1*pointEval;
454  kahanSum(eval, c, penalise);
455  }
456  ++HBCollisions;
457  }
458  }
459  //if (observation) printf("eval = %f\n", eval);
460  return eval;
461  }

485  template <typename _Ptr> inline _RealEval estimate(Mat34& pose, _Ptr begin, _Ptr end, _RealEval depthStdDev, size_t& HBCollisions, const _RealEval maxDist = numeric_const<_RealEval>::MAX) const {
486  _RealEval eval = numeric_const<_RealEval>::ZERO, c = numeric_const<_RealEval>::ZERO;
487  for (_Ptr i = begin; i < end; ++i) {
488  const Real distance = getDistance(pose, *i, maxDist);
489  if (distance > numeric_const<Real>::ZERO && distance < maxDist) {
490  const _RealEval pointEval = Math::exp(-depthStdDev*Real(distance));
491  kahanSum(eval, c, pointEval);
492  ++HBCollisions;
493  }
494  }
495  return eval;
496  }
