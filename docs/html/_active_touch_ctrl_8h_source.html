122  template <typename _Real> inline _Real N(const _Real x, const _Real stdev) {
123  const _Real norm = golem::numeric_const<_Real>::ONE / (stdev*golem::Math::sqrt(2 * golem::numeric_const<_Real>::PI));
124  return norm*golem::Math::exp(-.5*golem::Math::sqr(_Real(x) / _Real(stdev))); // gaussian
125  };

127  template <typename _Ptr, typename _Real> inline std::vector<_Real> N(_Ptr begin, _Ptr end, const size_t dim, const _Real stdev) {
128  std::vector<_Real> output;
129  output.assign(dim, golem::numeric_const<_Real>::ZERO);
130  size_t idx = 0;
131  for (_Ptr i = begin; i != end; ++i) {
132  output[idx++] = N(*i, stdev);
133  }
