160  template <typename _Val, typename _Cov> static void getProperty(size_t size, size_t dim, _Val val, _Cov cov, Property& property) {
161  if (size <= 0 || dim <= 0)
162  throw golem::Message(golem::Message::LEVEL_ERROR, "Feature3D::getCovariance(): no data provided");
163 
164  // estimate bandwidth - scott04multi-dimensional
165  const Real bandwidthSqr = golem::Math::pow(golem::numeric_const<Real>::ONE/size, golem::numeric_const<Real>::TWO/(dim + 4));
166 
167  // compute covariance
168  property.mean.resize(dim);
169  property.covariance.resize(dim);
170  property.covarianceSqrt.resize(dim);
171  property.covarianceInv.resize(dim);
172  property.covarianceInvSqrt.resize(dim);
173  if (!property.create(size, dim, val, [=] (size_t j, Real val) -> Real { return cov(j, bandwidthSqr*val); }))
174  throw golem::Message(golem::Message::LEVEL_ERROR, "Feature3D::getCovariance(): Unable to create covariance property");
175  }

178  static std::string to_string(const Feature& feature, const char* format, bool index = false) {
179  char buff[BUFSIZ], *ptr = buff;
180  *ptr = '\0';
181  for (size_t i = 0; i < feature.size(); ++i)
182  ptr = index ? golem::snprintf(ptr, buff + sizeof(buff), format, i + 1, feature[i]) : golem::snprintf(ptr, buff + sizeof(buff), format, feature[i]);
183  return std::string(buff);
184  }
