787  template <typename _Ptr> inline static golem::Real evaluateSample(_Ptr begin, _Ptr end, const golem::RBCoord& coord) {
788  golem::Real likelihood = golem::numeric_const<golem::Real>::ZERO, c = golem::numeric_const<golem::Real>::ZERO;
789  for (_Ptr kernel = begin; kernel < end; ++kernel) {
790  const golem::Real dlin = kernel->p.distanceSqr(coord.p);
791  if (dlin < kernel->distMax.lin) {
792  const golem::Real dang = kernel->q.distance(coord.q);
793  if (dang < kernel->distMax.ang) {
794  const golem::Real distance = kernel->covInv.lin*dlin + kernel->covInv.ang*dang;
795  const golem::Real sampleLikelihood = kernel->weight*golem::Math::exp(-golem::Real(distance));
796  golem::kahanSum(likelihood, c, sampleLikelihood);
797  }
798  }
799  }
800  return likelihood;
801  }
