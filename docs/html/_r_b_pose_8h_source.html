138  const golem::Real d1 = golem::REAL_ONE - (a[0]*b[0] + a[1]*b[1] + a[2]*b[2]);

564  template <typename _Seq> void createFeature(const _Seq& points, Feature::Seq& features) const {
565  if (points.size() < 2)
566  throw golem::Message(golem::Message::LEVEL_ERROR, "RBPose::createFeature(): Insufficient number of points");
567 
568  size_t attempt = 0;
569  for (Feature::Seq::iterator i = features.begin(); i != features.end();) {
570  if (++attempt > desc.attempts)
571  throw golem::Message(golem::Message::LEVEL_ERROR, "RBPose::createFeature(): Unable to generate features");
572 
573  const size_t pi1 = rand.next()%points.size();
574  const size_t pi2 = (pi1 + 1 + rand.next()%(points.size() - 1))%points.size();
575  if (!createFeature(points[pi1], points[pi2], *i))
576  continue;
577 
578  i->index[0] = pi1;
579  i->index[1] = pi2;
580  ++i;
581  attempt = 0;
582  }
