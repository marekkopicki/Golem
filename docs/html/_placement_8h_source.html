755  template <typename _Ptr> inline static golem::Real evaluateSample(_Ptr begin, _Ptr end, const golem::RBCoord& coord) {
756  golem::Real likelihood = golem::numeric_const<golem::Real>::ZERO, c = golem::numeric_const<golem::Real>::ZERO;
757  for (_Ptr kernel = begin; kernel < end; ++kernel) {
758  const golem::Real dlin = kernel->p.distanceSqr(coord.p);
759  if (dlin < kernel->distMax.lin) {
760  const golem::Real dang = kernel->q.distance(coord.q);
761  if (dang < kernel->distMax.ang) {
762  const golem::Real distance = kernel->covInv.lin*dlin + kernel->covInv.ang*dang;
763  const golem::Real sampleLikelihood = kernel->weight*golem::Math::exp(-golem::Real(distance));
764  golem::kahanSum(likelihood, c, sampleLikelihood);
765  }
766  }
767  }
768  return likelihood;
769  }
