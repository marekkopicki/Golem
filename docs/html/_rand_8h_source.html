359  template <typename Type, typename Pointer> inline void nextNVariable(Pointer begin, Pointer end) const {
360  // generate random N-dimensional point contained in a N-dimensional ball of radius 1
361  // this rejection sampling becomes inefficient for large N:
362  // probability of rejection is proportional to 1 - V_N/2^N where V_N is the volume of N-dimensional ball of radius 1
363  // V_N = 2*pi*R^2/N * V_(N-2), V_1 = 1; V_2 = 2*R and R = 1;
364  Type s;
365  do {
366  s = numeric_const<Type>::ZERO;
367  for (Pointer i = begin; i != end; ++i)
368  s += Math::sqr(*i = nextUniform<Type>()*numeric_const<Type>::TWO - numeric_const<Type>::ONE);
369  } while (s > numeric_const<Type>::ONE - numeric_const<Type>::EPS || s < numeric_const<Type>::EPS);
370

372  for (Pointer i = begin; i != end; ++i)
