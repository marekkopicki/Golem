171  template <typename _Real> void getIntrinsicMatrix(_Real zNear, _Real zFar, _Real intrinsic[]) const {
172  // intrinsic parameters
173  // transform the coordinate system of computer vision to OpenGL
174  // Vision: origin is in the up left corner, x-axis pointing right, y-axis pointing down
175  // OpenGL: origin is in the middle, x-axis pointing right, y-axis pointing up
176 
177  // scale range from [0 ... width] to [0 ... 2]
178  const _Real fx = (_Real)(golem::REAL_TWO*this->fx/this->width);
179  // scale range from [0 ... height] to [0 ... 2]
180  const _Real fy = (_Real)(golem::REAL_TWO*this->fy/this->height);
181  // move coordinates from left to middle of image: [0 ... 2] -> [-1 ... 1] (not negative z value at w-division)
182  const _Real cx = (_Real)(golem::REAL_ONE - (golem::REAL_TWO*this->cx/this->width));
183  // flip and move coordinates from top to middle of image: [0 ... 2] -> [-1 ... 1] (not negative z value at w-division)
184  const _Real cy = (_Real)((golem::REAL_TWO*this->cy/this->height) - golem::REAL_ONE);
185  // entries for clipping planes
186  const _Real z1 = (zFar + zNear)/(zNear - zFar);
187  // look up for gluPerspective
188  const _Real z2 = _Real(2.0)*zFar*zNear/(zNear - zFar);
189 
190  // intrinsic matrix
191  // last row assigns w=-z which inverts cx and cy at w-division
192  intrinsic[0] = fx; intrinsic[4] = 0; intrinsic[8] = cx; intrinsic[12] = 0;
193  intrinsic[1] = 0; intrinsic[5] = fy; intrinsic[9] = cy; intrinsic[13] = 0;
194  intrinsic[2] = 0; intrinsic[6] = 0; intrinsic[10] = z1; intrinsic[14] = z2;
195  intrinsic[3] = 0; intrinsic[7] = 0; intrinsic[11] = -1; intrinsic[15] = 0;
196  }
198  template <typename _Real> static void getExtrinsicMatrix(const golem::Mat34& pose, _Real extrinsic[]) {
199  // extrinsic matrix
200  golem::Mat34 invPose, ex, flipYZ;
201  invPose.setInverseRT(pose);
202  flipYZ.setId();
203  flipYZ.R.m[1][1] = -1.0;
204  flipYZ.R.m[2][2] = -1.0;
205  ex.multiply(flipYZ, invPose);
206  ex.getColumn44(extrinsic);
207  }
