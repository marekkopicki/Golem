129  template <typename _Handler> void save(const std::string& path, _Handler handler) {
130  // new path
131  const bool hasNewPath = this->path != path;
132  if (isModified()) {
133  // always save if modified
134  handler(path);
135  setModified(false);
136  // new path
137  if (hasNewPath) {
138  // delete old file
139  if (desc.deleteIfMoved) remove(this->path);
140  // update current path
141  this->path = path;
142  }
143  }
144  else if (hasNewPath) {
145  if (boost::filesystem::exists(boost::filesystem::path(this->path))) {
146  if (desc.deleteIfMoved) {
147  // there exists a new path - remove it
148  if (boost::filesystem::exists(boost::filesystem::path(path))) remove(path);
149  // path is different - move
150  boost::filesystem::rename(boost::filesystem::path(this->path), boost::filesystem::path(path));
151  }
152  else {
153  // path is different - copy
154  boost::filesystem::copy_file(boost::filesystem::path(this->path), boost::filesystem::path(path), boost::filesystem::copy_option::overwrite_if_exists);
155  }
156  }
157  // update current path
158  this->path = path;
159  }
160  }
162  void remove() {
163  if (desc.deleteIfUnlinked) {
164  remove(path);
165  path.clear();
166  }
167  }
