100  static inline Real getDepth(const typename Triangle::Seq& triangles, const Vec3& point, Real offset) {
101  Real depth = golem::numeric_const<Real>::MAX;
102  for (typename Triangle::Seq::const_iterator i = triangles.begin(); i != triangles.end(); ++i) {
103  const Real d = i->distance + offset - i->normal.dot(point);
104  if (d < golem::numeric_const<Real>::ZERO) // no collision
105  return d;
106  if (depth > d) // search for minimum
107  depth = d;
108  }
109  return depth;
110  }
112  inline Real getDepth(const Vec3& point, Real offset) const {
113  Real depth = golem::numeric_const<Real>::ZERO; // if no bounds or collisions, no effect
114  for (typename Triangle::SeqSeq::const_iterator i = triangles.begin(); i != triangles.end(); ++i) {
115  const Real d = getDepth(*i, point, offset);
116  if (depth < d) // search for maximum depth (can collide only with one mesh within bounds)
117  depth = d;
118  }
119  return depth;
120  }

124  template <typename _Ptr> inline void evaluate(_Ptr begin, _Ptr end, _RealEval depthOffset, _RealEval depthStdDev, _RealEval& eval, _RealEval& depth, size_t& collisions) const {
125  for (_Ptr i = begin; i < end; ++i) {
126  const Real pointDepth = getDepth(*i, depthOffset);
127  if (pointDepth > golem::numeric_const<Real>::ZERO) {
128  const _RealEval pointEval = golem::Math::exp(depthStdDev*_RealEval(pointDepth)) - golem::numeric_const<Real>::ONE;
129  eval += pointEval;
130  depth += pointDepth;
131  ++collisions;
132  }
133  }
