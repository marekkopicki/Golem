210  inline void fromScrew(Real h, const Vec3& anchor, const Vec3& axis) {
211  if (Math::isFinite(h)) {
212  // Rotation and translation
213  // [ v, w ] = [ -axis x anchor + h axis = anchor x axis + h axis, axis ]
214  v.cross(anchor, axis);
215  w.multiply(h, axis); // temp
216  v.add(v, w);
217  w = axis;
218  }
219  else {
220  // Pure translation
221  // [ v, w ] = [ axis, 0 ]
222  v = axis;
223  w.setZero();
224  }
225  }
