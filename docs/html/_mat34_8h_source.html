28 template <typename _Mat34, typename _Twist, typename _Real> inline void TwistToMat34(_Mat34& m, const _Twist& t, _Real theta) {
29  typedef _Real Real;
30  typedef typename _Mat34::Vec3 Vec3;
31  typedef typename _Mat34::Mat33 Mat33;
32 
33  if (t.w.isZero()) {
34  // [ Id, v * theta ]
35  m.R.setId();
36  m.p.multiply(theta, t.v);
37  }
38  else {
39  // [ exp(w^ theta), [Id - exp(w^ theta)] w^ v + w wT v * theta ]
40  const Real c1 = t.w.v2*t.v.v3 - t.w.v3*t.v.v2;
41  const Real c2 = t.w.v3*t.v.v1 - t.w.v1*t.v.v3;
42  const Real c3 = t.w.v1*t.v.v2 - t.w.v2*t.v.v1;
43 
44  const Real s = t.w.dot(t.v) * theta;
45 
46  m.R.fromAngleAxis(theta, t.w);
47 
48  m.p.v1 = (numeric_const<Real>::ONE - m.R.m11) * c1 - m.R.m12 * c2 - m.R.m13 * c3 + t.w.v1 * s;
49  m.p.v2 = -m.R.m21 * c1 + (numeric_const<Real>::ONE - m.R.m22) * c2 - m.R.m23 * c3 + t.w.v2 * s;
50  m.p.v3 = -m.R.m31 * c1 - m.R.m32 * c2 + (numeric_const<Real>::ONE - m.R.m33) * c3 + t.w.v3 * s;
51  }
52 }

54 template <typename _Mat34, typename _Twist, typename _Real> inline void Mat34ToTwist(_Twist& t, _Real& theta, const _Mat34& m) {
55  typedef _Real Real;
56  typedef typename _Mat34::Vec3 Vec3;
57  typedef typename _Mat34::Mat33 Mat33;
58 
59  m.R.toAngleAxis(theta, t.w);
60 
61  if (Math::abs(theta) < numeric_const<Real>::EPS) { // see Mat33::toAngleAxis()
62  // no rotation
63  theta = m.p.magnitude();
64 
65  if (Math::abs(theta) > numeric_const<Real>::EPS)
66  // no translation
67  t.v.multiply(numeric_const<Real>::ONE / theta, m.p);
68  else
69  t.v.setZero();
70 
71  t.w.setZero();
72  }
73  else {
74  // A * v = p => v = inv(A) * p, where A := [Id - exp(w^ theta)] w^ + w wT * theta
75  // A is nonsingular for all theta E (0, 2*PI), because matrices which comprise A
76  // have mutually orthogonal null spaces for theta != 0 and w != 0 (R != Id)
77  Mat33 temp;
78 
79  temp.m11 = m.R.m13*t.w.v2 - m.R.m12*t.w.v3 + t.w.v1*t.w.v1*theta;
80  temp.m12 = -m.R.m13*t.w.v1 - (numeric_const<Real>::ONE - m.R.m11)*t.w.v3 + t.w.v1*t.w.v2*theta;
81  temp.m13 = (numeric_const<Real>::ONE - m.R.m11)*t.w.v2 + m.R.m12*t.w.v1 + t.w.v1*t.w.v3*theta;
82 
83  temp.m21 = (numeric_const<Real>::ONE - m.R.m22)*t.w.v3 + m.R.m23*t.w.v2 + t.w.v2*t.w.v1*theta;
84  temp.m22 = m.R.m21*t.w.v3 - m.R.m23*t.w.v1 + t.w.v2*t.w.v2*theta;
85  temp.m23 = -m.R.m21*t.w.v2 - (numeric_const<Real>::ONE - m.R.m22)*t.w.v1 + t.w.v2*t.w.v3*theta;
86 
87  temp.m31 = -m.R.m32*t.w.v3 - (numeric_const<Real>::ONE - m.R.m33)*t.w.v2 + t.w.v3*t.w.v1*theta;
88  temp.m32 = (numeric_const<Real>::ONE - m.R.m33)*t.w.v1 + m.R.m31*t.w.v3 + t.w.v3*t.w.v2*theta;
89  temp.m33 = m.R.m32*t.w.v1 - m.R.m31*t.w.v2 + t.w.v3*t.w.v3*theta;
90
