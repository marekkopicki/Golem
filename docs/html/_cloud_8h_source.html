1366  golem::CriticalSection cs;
1367  ParallelsTask(context.getParallels(), [&] (ParallelsTask*) {

1369  for (size_t j;;) {

1372  if (i*threadChunkSize > seq.size()) {
1373  inpSize += size;

1375  break;

1377  j = i++;
1378  }
1379  for (typename _Seq::iterator k = seq.begin() + j*threadChunkSize, end = seq.begin() + std::min((j + 1)*threadChunkSize, seq.size()); k != end; ++k)

1408  throw golem::Message(golem::Message::LEVEL_ERROR, "Cloud::filter(): no input point clouds");
1409  for (typename _Seq::const_iterator k = inp.begin(); k != inp.end(); ++k)
1410  if (inp.begin()->size() != k->size())
1411  throw golem::Message(golem::Message::LEVEL_ERROR, "Cloud::filter(): input point clouds sizes does not match");
1412  // resize
1413  if (out.size() != inp.begin()->size())
1414  out.resize(inp.begin()->size());
1415  out.width = inp.begin()->width;
1416  out.height = inp.begin()->height;
1417  // sensor origins
1418  typedef std::vector<pcl::PointXYZ> PointXYZSeq;
1419  PointXYZSeq originSeq;
1420  for (typename _Seq::const_iterator k = inp.begin(); k != inp.end(); ++k)
1421  originSeq.push_back(pcl::PointXYZ(k->sensor_origin_.x(), k->sensor_origin_.y(), k->sensor_origin_.z()));
1422  // run filter
1423  typedef std::vector<std::pair<size_t, float> > DepthRank;
1424  auto depthRankCmp = [] (const DepthRank::value_type& l, const DepthRank::value_type& r) -> bool { return l.first < r.first; };
1425  const size_t samples = std::min(size_t(desc.samples), inp.size());
1426  size_t i = 0;
1427  golem::CriticalSection cs;
1428  ParallelsTask(context.getParallels(), [&] (ParallelsTask*) {
1429  DepthRank depthRank;
1430  for (size_t l;;) {
1431  {
1432  golem::CriticalSectionWrapper csw(cs);
1433  if (i*threadChunkSize > out.size())
1434  break;
1435  l = i++;
1436  }
1437  for (size_t j = l*threadChunkSize, end = std::min((l + 1)*threadChunkSize, out.size()); j < end; ++j) {
1438  depthRank.clear();
1439  for (size_t k = 0; k < inp.size(); ++k) {
1440  const Point point = (*(inp.begin() + k))[j];
1441  if (isNanXYZ(point))
1442  continue;
1443  const pcl::PointXYZ origin = originSeq[k];
1444  const float d = golem::Math::sqr(point.x - origin.x) + golem::Math::sqr(point.y - origin.y) + golem::Math::sqr(point.z - origin.z);
1445  depthRank.push_back(std::make_pair(k, d));
1446  }
1447  if (depthRank.size() < desc.samples) {
1448  setNanXYZ(out[j]);
1449  continue;
1450  }
1451  const size_t element = depthRank.size()/2;
1452  std::nth_element(depthRank.begin(), depthRank.begin() + element + 1, depthRank.end(), depthRankCmp);
1453  out[j] = (*(inp.begin() + depthRank[element].first))[j];
1454  }
1455  }
