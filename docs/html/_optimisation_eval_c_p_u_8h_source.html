89  this->kernels.clear();
90  this->kernels.reserve(kernels.size());
91  for (Configuration::Kernel::Seq::const_iterator i = kernels.begin(); i != kernels.end(); ++i)
92  this->kernels.push_back(Kernel(begin, end, *i));
93 
94  if (kernels.back().cdf < golem::numeric_const<golem::Real>::EPS)
95  throw golem::Message(golem::Message::LEVEL_ERROR, "EvalConfigModelCPU::create(): kernels are not normalised");
96  dist = Config(begin, end, space.desc.configCovInv.data());
97  distMax = static_cast<_Real>(golem::Math::sqr(space.desc.distanceStdDevMax));
98  weight = static_cast<_RealEval>(golem::numeric_const<golem::Real>::ONE / kernels.back().cdf);

101  inline _RealEval evaluate(const Config& config) const {
102  _RealEval likelihood = golem::numeric_const<_RealEval>::ZERO, c = golem::numeric_const<_RealEval>::ZERO;
103  for (const Kernel *i = kernels.data(), *end = i + kernels.size(); i < end; ++i) {
104  const _Real distance = i->distance(size, config, dist);
105  if (distance < distMax) {
106  const _RealEval sampleLikelihood = i->weight*golem::Math::exp(-_RealEval(distance));
107  golem::kahanSum(likelihood, c, sampleLikelihood);
108  }
109  }
110  return weight*likelihood;
