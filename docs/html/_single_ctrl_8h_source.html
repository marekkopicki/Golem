365  template <typename _Ptr> _Ptr send(_Ptr begin, _Ptr end, bool clear = false, bool limits = true, MSecTmU32 timeWait = MSEC_TM_U32_INF) {
366  _Ptr ptr = begin;
367 
368  if (begin != end) {
369  SysTimer timer;
370 
371  // assert that trajectory segments are in the limits
372  if (limits)
373  for (_Ptr i = begin, j = begin; ++j != end; ++i)
374  assertLimits(*i, *j);
375 
376  SecTmReal shift;
377 
378  {
379  CriticalSectionWrapper csw(csCommand);
380 
381  // compute time shift if sent trajectory segments time stamps corresponds to past
382  shift = std::max(commandTime + cycleDuration - ptr->t, SEC_TM_REAL_ZERO);
383  // find trajectory segments to remove
384  const Queue::iterator last = std::lower_bound(qCommand->begin(), clear ? ++qCommand->begin() : qCommand->end(), ptr->t + shift);
385  // regardless of limits flag, assert that the first trajectory segment is in the limits
386  if (!qCommand->empty() && last != qCommand->begin()) {
387  State state = createState(); // range from this controller
388  state = *ptr; // data copying, no range acquisition from ptr
389  state.t += shift;
390  if (limits)
391  assertLimits(*(last - 1), state);
392  }
393  // clear events
394  evCommandBegin.set(false);
395  evCommandEnd.set(false);
396  // erase all the trajectory waypoints with time stamps larger than begin->t
397  qCommand->erase(last, qCommand->end());
398  // send as much segments as possible
399  for (size_t capacity = qCommand->capacity() - qCommand->size(); capacity > 0 && ptr != end; ++ptr, --capacity) {
400  qCommand->push_back(*ptr); // copy constructor, no range acquisition
401  qCommand->back().t += shift;
402  }
403  }
404 
405  // queue up next segments if there are any left and if there is time left
406  if (timeWait > MSEC_TM_U32_ZERO)
407  while (ptr != end) {
408  const MSecTmU32 elapsed = timer.elapsed();
409  if (!waitForBegin(timeWait < elapsed ? MSEC_TM_U32_ZERO : timeWait - elapsed))
410  return ptr;
411 
412  CriticalSectionWrapper csw(csCommand);
413 
414  // if the queue is still full, try again
415  if (qCommand->full())
416  continue;
417 
418  qCommand->push_back(*ptr);
419  qCommand->back().t += shift;
420  ++ptr;
421  }
422  }
423 
424  // inform the associated interface
425  getCallbackDataSync()->syncSend(&*begin);
426 
427  return ptr;
428  }
